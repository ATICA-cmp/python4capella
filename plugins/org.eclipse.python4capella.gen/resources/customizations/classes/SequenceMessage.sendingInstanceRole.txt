    def get_sending_instance_role(self):
        part =  self.get_java_object().getSendingPart()
        if part is None:
            return part
        else:
            value = part.getType()
            if value is None:
                return value
            else:
                e_object_class = getattr(sys.modules["__main__"], "EObject")
                specific_cls = e_object_class.get_class(value)
                return specific_cls(value)
